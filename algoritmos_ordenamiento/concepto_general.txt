	temp = p.getBurstRemainingTime()
	if temp < min_BTR:
		min_BTR = temp
		min_BTR_process = p
	elif (temp == min_BTR):
		Tie-Breaker.

os.InterruptHandler(RQ_TO_CPU, min_BTR_process) // Ejecutamos.
